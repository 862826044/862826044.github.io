HashMap的简单使用(Rust)

2021-04-08 
use std::collections::{HashMap, HashSet};

fn main() {

let mut hd: HashMap<&str, HashMap<&str, Box<Vec>>> = HashMap::new();
let mut tm1 = HashMap::new();
//方法一： 如果tm1没有“book1”key,则insert key为“book1”和相应元素值。
tm1.entry(“book1”).or_insert(Box::new(vec![1.0_f32]));
hd.insert(“rust”, tm1);
println!(“{:?}”, hd);

//方法二： 直接insert
let mut tm2 = HashMap::new();
tm2.insert(“book1”, Box::new(vec![2.0_f32]));
hd.insert(“julia”, tm2);

println!(“{:?}”, hd);

// insert： Rust允许相同key(julia)的重复操作，会进行覆盖操作。

let mut tm3 = HashMap::new();
tm3.insert("book1", Box::new(vec![3.0_f32]));
tm3.insert("book2", Box::new(vec![4.0_f32]));
hd.insert("julia", tm3);
println!("{:?}", hd);


hd.get_mut("julia").unwrap().get_mut("book2").unwrap().push(4.0);
println!("{:?}", hd);

}
